#include<stdio.h>
struct membership{

int age;
char name[20];
float dom;
};
typedef struct membership membership;
int main()
{
membership m[50];
int i=0;
for (i=0;i<5;i++){
printf("enter name and age of %d person:",i+1);
scanf("%s %d",&m[i].name,&m[i].age);
printf("\n");
}
for (i=0;i<5;i++){
if (m[i].age<=20){
m[i].dom=1;
}else if(m[i].age<=30){
m[i].dom=(30.0-m[i].age)/10.0;
}else{
m[i].dom=0;
}
}
for (i=0;i<5;i++){
printf("name=%s\t age=%d\t degree of membership=%f\n",m[i].name,m[i].age,m[i].dom);
}
return 0;
}
